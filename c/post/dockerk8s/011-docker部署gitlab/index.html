#!/bin/bash
GITLAB_DIR=/home/lucas/gitlab/data

docker run -d \
    -p 5443:443 \
    -p 9998:9998 \
    -p 2222:22 \
    --name gitlab \
    -v ${GITLAB_DIR}/config:/etc/gitlab \
    -v ${GITLAB_DIR}/logs:/var/log/gitlab \
    -v ${GITLAB_DIR}/data:/var/opt/gitlab \
    gitlab/gitlab-ce:latest
